    function resolverEquacao2Grau(a, b, c) {
    if (a === 0) {
        return "O coeficiente 'a' não pode ser zero (equação do 1º grau).";
    }

    // Cálculo do delta: b^2 - 4ac
    const delta = Math.pow(b, 2) - 4 * a * c;

    if (delta < 0) {
      return "A equação não possui raízes reais (delta negativo).";
    } 
    else if (delta === 0) {
     // Apenas uma raiz real
    const x = -b / (2 * a);
    return `A equação possui uma raiz real: x = ${x}`;
    } 
    else {
    // Duas raízes reais
         const x1 = (-b + Math.sqrt(delta)) / (2 * a);
        const x2 = (-b - Math.sqrt(delta)) / (2 * a);
        return `Raízes: x1 = ${x1.toFixed(2)}, x2 = ${x2.toFixed(2)}`;
    }
    }

    // Exemplo de uso:
    // Equação: 1x^2 - 5x + 6 = 0
    console.log(resolverEquacao2Grau(1, -5, 6)); // Saída: Raízes: x1 = 3.00, x2 = 2.00
